# Vercel 环境变量配置模板
# 在Vercel项目设置中添加以下环境变量

# 基本配置
NODE_ENV=production

# MongoDB 连接配置
# 请替换为您的实际MongoDB连接信息
MONGODB_USERNAME=your_mongodb_username
MONGODB_PASSWORD=your_mongodb_password
MONGODB_CLUSTER=your_mongodb_cluster
MONGODB_DATABASE=leshan_furniture
MONGODB_COLLECTION=prices

# 服务器配置
PORT=3000
HOST=0.0.0.0
